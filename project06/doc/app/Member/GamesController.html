def create
  @game = Game.new(params[:game])
  @game.user_id = current_user.id

  respond_to do |format|
    if @game.save
      format.html { redirect_to member_root_url, :notice => "Successfully added game" }
      format.json { render :json => @game, :status => :created, :location => @game }
    else
      format.html { render :action => "new" }
      format.json { render :json => @game.errors, :status => :unprocessable_entity }
    end
  end
end

def index
  @games = Game.paginate(:page => params[:page], :order => 'created_at desc', :per_page => 10).where(:user_id => current_user.id)

  respond_to do |format|
    format.html # index.html.erb
    format.json { render :json => @games }
  end
end

def update
  @game = Game.find(params[:id])

  respond_to do |format|
    if @game.update_attributes(params[:game])
      format.html { redirect_to member_root_url, :notice => "Successfully updated game" }
      format.json { head :no_content }
    else
      format.html { render :action => "edit" }
      format.json { render :json => @game.errors, :status => :unprocessable_entity }
    end
  end
end
